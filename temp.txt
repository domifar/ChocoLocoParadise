<script>
    var numberLine = document.getElementById('numberLine');
    var redSection = document.getElementById('redSection');
    var greenSection = document.getElementById('greenSection');
    var marker = document.getElementById('marker');
    var point = document.getElementById('point');
    var isRedFirst = true;

    function placePoint(position) {
        point.style.left = position + '%';
        point.style.display = 'block';
    }

    function placeMarker(position) {
        marker.style.left = position + '%';
        marker.style.display = 'block';
        setColors(position);
    }

    function setColors(position) {
        if (isRedFirst) {
            redSection.style.right = (100 - position) + '%';
            greenSection.style.left = position + '%';
        } else {
            greenSection.style.right = (100 - position) + '%';
            redSection.style.left = position + '%';
        }
    }

    function swapColors() {
        isRedFirst = !isRedFirst;
        var position = parseInt(marker.style.left);
        setColors(position);
    }
</script>

</body>

    var numberLine = document.getElementById('numberLine');
    var marker = document.getElementById('marker');
    var point = document.getElementById('point');
    var isRedFirst = true;

    function placePoint(position) {
        point.style.left = position + '%';
        point.style.display = 'block';
    }

    function placeMarker(position) {
        marker.style.left = position + '%';
        marker.style.display = 'block';
        setColors(position);

    function setColors(position) {
        if (isRedFirst) {
            numberLine.style.background = 'linear-gradient(to right, red ' + position + '%, green ' + position + '%)';

        isRedFirst = !isRedFirst;
